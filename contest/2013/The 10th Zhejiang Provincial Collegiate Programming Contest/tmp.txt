using namespace std;

struct T{
  int c,pre,pos;
  bool operator<(T const& a, T const &b){
    return a.c > b.c;
  }
}str[555];

int num[555];

int main(){

    int n,p,i,now_val,now_pos;
    int cas,ans;
    scanf("%d",&cas);
    while(cas--){
        scanf("%d",&n);
        memset(num,0,sizeof(num));
        for(i=2;i<=n;i++){
            scanf("%d",&str[i].pre);
            str[i].pos = 2;
        }
        p=n-1;
        ans = 0;
        for(i=2;i<=n;i++){
            scanf("%d",&str[i].c);
            ans += str[i].c;
        }
        sort(str+2,str+n+1);
        now_pos = 2;
        now_val = 0;
        for(i=2;i<=n;i++){
            printf("%d  %d  %d\n",str[i].c,str[i].pre,str[i].pos);
            ans -= str[i].c;
            p--;
            num[str[i].pre]++;
            if(num[str[i].pre] >= now_val){
                now_val = num[str[i].pre];
                now_pos = i;
            }
            if(p < num[str[i].pre]){
                p++;
                ans += str[i].c;
                break;
            }
        }

        printf("%d",ans);
    }

    return 0;
}
