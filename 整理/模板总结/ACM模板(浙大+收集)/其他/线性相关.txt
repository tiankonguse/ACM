//判线性相关(正交化)
//传入m个n维向量
#include <math.h>
#define MAXN 100
#define eps 1e-10

int linear_dependent(int m,int n,double vec[][MAXN]){
    double ort[MAXN][MAXN],e;
    int i,j,k;
    if (m>n)
        return 1;
    for (i=0;i<m;i++){
        for (j=0;j<n;j++)
            ort[i][j]=vec[i][j];
        for (k=0;k<i;k++){
            for (e=j=0;j<n;j++)
                e+=ort[i][j]*ort[k][j];
            for (j=0;j<n;j++)
                ort[i][j]-=e*ort[k][j];
            for (e=j=0;j<n;j++)
                e+=ort[i][j]*ort[i][j];
            if (fabs(e=sqrt(e))<eps)
                return 1;
            for (j=0;j<n;j++)
                ort[i][j]/=e;
        }
    }
    return 0;
}
